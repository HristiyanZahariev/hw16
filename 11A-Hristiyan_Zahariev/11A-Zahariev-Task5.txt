function pl3d
  z = 0:pi/200:2; #Създавам масив започващ от 0 до 2 със стъпка pi/200
  x = cos(z); #Създавам масив приемащ стойността на cos(z)
  y = sin(z); #Създавам масив приемащ стойността на sin(z)
  plot3(x,y,z,"+"); #Чертаем графиката, използвам plot3, защото използваме триизмерна координатна система
endfunction
